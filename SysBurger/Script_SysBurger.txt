CREATE TABLE [dbo].[TB_PESSOA] 
(
    [I_COD_PESSOA]  INT IDENTITY (1, 1) NOT NULL,
    [B_DEL_PESSOA]  TINYINT NOT NULL,
    [S_NM_PESSOA]   VARCHAR (35) NOT NULL,
    [S_SNM_PESSOA]  VARCHAR (35) NOT NULL,
    [S_CPF_PESSOA]  VARCHAR (14) NOT NULL,
    [S_CEL_PESSOA]  VARCHAR (16) NOT NULL,
    [S_MAIL_PESSOA] VARCHAR (100) NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_PESSOA] ASC)
);

CREATE TABLE [dbo].[TB_ENDERECO] 
(
    [I_COD_ENDERECO] INT          IDENTITY (1, 1) NOT NULL,
    [I_COD_PESSOA]   INT          NOT NULL,
    [S_END_ENDERECO] VARCHAR (30) NOT NULL,
    [S_BAI_ENDERECO] VARCHAR (25) NOT NULL,
    [S_CID_ENDERECO] VARCHAR (25) NOT NULL,
    [S_UF_ENDERECO]  VARCHAR (2)  NOT NULL,
    [S_CEP_ENDERECO] VARCHAR (10) NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_ENDERECO] ASC),
    FOREIGN KEY ([I_COD_PESSOA]) REFERENCES [dbo].[TB_PESSOA] ([I_COD_PESSOA])
);

CREATE TABLE [dbo].[TB_INGREDIENTE] 
(
    [I_COD_INGREDIENTE] INT IDENTITY (1, 1) NOT NULL,
    [B_DEL_INGREDIENTE] TINYINT NOT NULL,
    [S_NM_INGREDIENTE]  VARCHAR (30) NOT NULL,
    [I_QNT_INGREDIENTE] INT NOT NULL,
    [B_ATV_INGREDIENTE] TINYINT NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_INGREDIENTE] ASC)
);

CREATE TABLE [dbo].[TB_PRODUTO] 
(
    [I_COD_PRODUTO]     INT IDENTITY (1, 1) NOT NULL,

    PRIMARY KEY CLUSTERED ([I_COD_PRODUTO] ASC)
);


CREATE TABLE [dbo].[TB_INGREDIENTE_PRODUTO] 
(
    [I_COD_INGREDIENTE_PRODUTO]   INT IDENTITY (1, 1) NOT NULL,
    [I_COD_PRODUTO]               INT NOT NULL,
    [I_COD_INGREDIENTE]   INT NOT NULL,
    [I_QNTUT_INGREDIENTE_PRODUTO] INT NOT NULL,
    PRIMARY KEY CLUSTERED ([I_COD_INGREDIENTE_PRODUTO] ASC),

    FOREIGN KEY ([I_COD_INGREDIENTE]) REFERENCES [dbo].[TB_INGREDIENTE] ([I_COD_INGREDIENTE])
);
